99def init( 

100 resume_cache: Optional[str] = None, 

101 update_config_hook: Optional[Callable] = None, 

102) -> None: 

103 """Initialize APPL with dotenv and config files. 

104 

105 Args: 

106 resume_cache: Path to the trace file used as resume cache. Defaults to None. 

107 update_config_hook: A hook to update the configs. Defaults to None. 

108 """ 

109 with global_vars.lock: 

110 # only initialize once 

111 if global_vars.initialized: 

112 logger.warning("APPL has already been initialized, ignore") 

113 return 

114 global_vars.initialized = True 

115 

116 now = pendulum.instance(datetime.datetime.now()) 

117 # Get the previous frame in the stack, i.e., the one calling this function 

118 frame = inspect.currentframe() 

119 if frame and frame.f_back: 

120 caller_path = frame.f_back.f_code.co_filename # Get file_path of the caller 

121 caller_funcname = frame.f_back.f_code.co_name # Get function name of the caller 

122 caller_basename = os.path.basename(caller_path).split(".")[0] 

123 caller_folder = os.path.dirname(caller_path) # Get folder of the caller 

124 caller_folder = get_folder(caller_folder) 

125 dotenvs = find_files(caller_folder, [".env"]) 

126 appl_config_files = find_files( 

127 caller_folder, ["appl.yaml", "appl.yml", "appl.json", "appl.toml"] 

128 ) 

129 # load dotenvs and appl configs from outer to inner with override 

130 for dotenv in dotenvs[::-1]: 

131 load_dotenv(dotenv, override=True) 

132 logger.info("Loaded dotenv from {}".format(dotenv)) 

133 for config_file in appl_config_files[::-1]: 

134 override_configs = load_config(config_file) 

135 logger.info("Loaded configs from {}".format(config_file)) 

136 configs.update(override_configs) 

137 if configs.getattrs("settings.logging.display.configs_update"): 

138 logger.info(f"update configs:\n{yaml.dump(override_configs.to_dict())}") 

139 else: 

140 caller_basename, caller_funcname = "appl", "<module>" 

141 dotenvs, appl_config_files = [], [] 

142 logger.error( 

143 "Cannot find the caller of appl.init(), fail to load .env and appl configs" 

144 ) 

145 

146 if update_config_hook: 

147 update_config_hook(configs) 

148 log_format = configs.getattrs("settings.logging.format") 

149 log_level = configs.getattrs("settings.logging.log_level") 

150 log_file = configs.getattrs("settings.logging.log_file") 

151 # set logger level for loguru 

152 logger.remove() # Remove default handler 

153 logger.add(sys.stderr, level=log_level, format=_get_loguru_format()) 

154 if log_file.get("enabled", False): 

155 if (log_file_format := log_file.get("path_format", None)) is not None: 

156 log_file_path = ( 

157 log_file_format.format( 

158 basename=caller_basename, funcname=caller_funcname, time=now 

159 ) 

160 + ".log" 

161 ) 

162 log_file.path = log_file_path 

163 file_log_level = log_file.get("log_level", None) or log_level 

164 logger.info(f"Logging to file: {log_file_path} with level {file_log_level}") 

165 # no need to overwrite the default format when writing to file 

166 logger.add(log_file_path, level=file_log_level, format=log_format) 

167 

168 configs["info"] = Configs( 

169 { 

170 "start_time": now.format("YYYY-MM-DD HH:mm:ss"), 

171 "dotenvs": dotenvs, 

172 "appl_configs": appl_config_files, 

173 } 

174 ) 

175 if configs.getattrs("settings.logging.display.configs"): 

176 logger.info(f"Using configs:\n{yaml.dump(configs.to_dict())}") 

177 

178 tracing = configs.getattrs("settings.tracing") 

179 strict_match = tracing.get("strict_match", True) 

180 if tracing.get("enabled", False): 

181 if (trace_file_format := tracing.get("path_format", None)) is not None: 

182 prefix = trace_file_format.format( 

183 basename=caller_basename, funcname=caller_funcname, time=now 

184 ) 

185 trace_file_path = f"{prefix}.pkl" 

186 meta_file = f"{prefix}_meta.json" 

187 tracing.trace_file = trace_file_path 

188 logger.info(f"Tracing file: {trace_file_path}") 

189 dump_file(configs.to_dict(), meta_file) 

190 global_vars.trace_engine = TraceEngine( 

191 trace_file_path, mode="write", strict=strict_match 

192 ) 

193 else: 

194 logger.warning("Tracing is enabled but no trace file is specified") 

195 

196 resume_cache = resume_cache or os.environ.get("APPL_RESUME_TRACE", None) 

197 if resume_cache: 

198 global_vars.resume_cache = resume_cache 

199 logger.info(f"Using resume cache: {resume_cache}") 

200 global_vars.resume_cache = TraceEngine( 

201 resume_cache, mode="read", strict=strict_match 

202 )
